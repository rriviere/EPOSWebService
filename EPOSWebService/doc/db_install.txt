DROP TABLE EPOSWEB.WS_NOTIFICATIONS;

CREATE TABLE EPOSWEB.WS_NOTIFICATIONS
(
 NOTIFICATION_ID INTEGER NOT NULL,
 ACTION VARCHAR(100),
 CATEGORY VARCHAR(100),
 DISPLAY_LAST_CHECKTIME VARCHAR(100),
 LAST_CHECK_TIME VARCHAR(100),
 STATUS VARCHAR(100),
 RESOLVED_TIME VARCHAR(100),
 PRIMARY KEY (NOTIFICATION_ID));

CREATE SEQUENCE EPOSWEB.WS_NOTIFICATION_SEQ AS INTEGER;

COMMIT;
